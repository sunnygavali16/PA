pragma solidity ^0.8.0;

contract B {

    mapping(address => uint256) b;

    function d(uint256 a) public {
        require(a > 0, "Amount must be greater than zero");
        b[msg.sender] = b[msg.sender] + a;
    }

    function w(uint256 a) public {
        require(a > 0, "Amount must be greater than zero");
        require(b[msg.sender] >= a, "Insufficient balance");
        b[msg.sender] = b[msg.sender] - a;
    }

    function showb() public view returns(uint256) {
        return b[msg.sender];
    }
}
